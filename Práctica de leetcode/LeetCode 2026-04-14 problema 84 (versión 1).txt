public class Solution {
    public int LargestRectangleArea(int[] heights) {
        int n = heights.Length;
        int[] left = new int[n];
        int[] right = new int[n];
        Stack<int> stack = new Stack<int>();

        // Por cada barra del histograma, se busca la posición de la primera barra a la izquierda
        // que sea más corta
        for(int i = 0; i < n; i++){
            while(stack.Count > 0 && heights[stack.Peek()] >= heights[i])
                stack.Pop();
            // Si es la barra más corta hasta ahora, la posición almacenada es -1
            left[i] = stack.Count <= 0 ? -1 : stack.Peek();
            stack.Push(i);
        }

        stack.Clear();

        // Por cada barra del histograma, se busca la posición de la primera barra a la derecha
        // que sea más corta
        for(int i = n - 1; i >= 0; i--){
            while(stack.Count > 0 && heights[stack.Peek()] >= heights[i]) 
                stack.Pop();
            right[i] = stack.Count <= 0 ? n : stack.Peek();
            stack.Push(i);
        }

        int largestArea = 0;

        for(int i = 0; i < n; i++){
            int width = right[i] - left[i] - 1;
            largestArea = Math.Max(largestArea, heights[i] * width);
        }

        return largestArea;
    }
}
