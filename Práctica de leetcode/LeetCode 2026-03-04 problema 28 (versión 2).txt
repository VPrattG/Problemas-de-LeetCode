public class Solution {
    public int StrStr(string haystack, string needle) {
        // Versión que no utiliza substring
        // Un poco más lenta, pero utiliza menos espacio
        if(haystack == needle || needle.Length == 0) return 0;
        if(needle.Length > haystack.Length) return -1;
        
        int need = needle.Length;
        
        for(int i = 0; i <= haystack.Length - need; i++){
            bool isMatch = true;
            
            // Revisa cada letra manualmente
            for(int j = 0; j < need; j++){
                if(haystack[i + j] != needle[j]){
                    isMatch = false;
                    break;
                }
            }
            
            if(isMatch) return i;
        }

        return -1;
    }
}
