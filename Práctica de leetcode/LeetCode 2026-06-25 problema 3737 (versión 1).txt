public class Solution {
    public int CountMajoritySubarrays(int[] nums, int target) {
        int n = nums.Length;
        int result = 0;

        // Permite explorar todos los subarreglos de izquierda a derecha
        // Solo simula crear los subarreglos
        for(int left = 0; left < n; left++){
            int targetCount = 0;

            for(int right = left; right < n; right++){
                if(nums[right] == target) targetCount++;

                int length = right - left + 1;

                // Que algo sea la mayoría requiere que sea más de la mitad
                // de un conjunto
                if(targetCount > length / 2) result++;
            }
        }

        return result;
    }
}
