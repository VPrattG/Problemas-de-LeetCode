public class Solution {
    public IList<string> FullJustify(string[] words, int maxWidth) {
        List<string> result = new List<string>();
        int start = 0;
        while(start < words.Length){
            int end = start;
            int lineLength = 0;

            // lineLength + words[end].Length + (end + start)
            // lineLength: caracteres actualmente en la línea
            // words[end].Length: longitud de la palabra que se va a agregar
            // (end - start) cantidad de espacios entre palabras
            while(end < words.Length && lineLength + words[end].Length + (end - start) <= maxWidth){
                lineLength += words[end].Length;
                end++;
            }

            // spaces es el total de espacios en la línea
            // gaps es el total de espacios entre palabras
            int spaces = maxWidth - lineLength;
            int gaps = end - start - 1;

            StringBuilder sb = new StringBuilder();

            // Casos para líneas con una sola palabra o la útlima línea
            if(end == words.Length || gaps == 0){
                for(int i = start; i < end; i++){
                    sb.Append(words[i]);
                    if(i < end - 1){
                        sb.Append(" ");
                    }
                }
                while(sb.Length < maxWidth){
                    sb.Append(" ");
                }
            } 
            // Líneas regulares
            else{
                int gapSpaces = spaces / gaps;
                int extraSpaces = spaces % gaps;

                for(int i = start; i < end; i++){
                    sb.Append(words[i]);
                    if(i < end - 1){
                        // Para agregar los espacios sobrantes, pero no de más
                        int addedSpaces = gapSpaces + (i - start < extraSpaces ? 1 : 0);
                        sb.Append(' ', addedSpaces);
                    }
                }
            }

            result.Add(sb.ToString());
            start = end;
        }
        
        return result;
    }
}
