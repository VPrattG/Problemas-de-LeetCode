public class Solution {
    public IList<int> GrayCode(int n) {
        // Desplazar el 1 n veces a la izquierda indica la cantidad de valores que se ven
        // Porque se mostrarán 2^n valores
        int amount = 1 << n;
        // Constructor que inicia la lista con una cantidad de espacios específica
        List<int> result = new List<int>(amount);
        
        for(int i = 0; i < amount; i++){
            // XOR + desplazamiento a la derecha permite reducir cambios a un solo bit
            result.Add(i ^ (i >> 1));
        }

        return result;
    }
}
