public class Solution {
    public int RemoveDuplicates(int[] nums) {
        if(nums.Length == 0) return 0;

        // Servirá como contador y posición, por eso el 1
        int uniqueNums = 1;

        // El primer valor siempre es único, por eso empieza en 1
        for(int i = 1; i < nums.Length; i++){
            if(nums[i] != nums[i - 1]){
                nums[uniqueNums++] = nums[i];
            }
        }

        return uniqueNums;
    }
}
