public class Solution {
    public int LadderLength(string beginWord, string endWord, IList<string> wordList) {
        HashSet<string> hs = new HashSet<string>(wordList);
        if(!hs.Contains(endWord)) return 0;

        // Candidatos para llegar a la última palabra
        Queue<string> queue = new Queue<string>();
        queue.Enqueue(beginWord);

        // Para no repetir palabras usadas previamente
        HashSet<string> visitedWords = new HashSet<string>();
        visitedWords.Add(beginWord);

        // Empieza en 1 porque se considera la primera palabra
        int changes = 1;

        // Uso de BFS
        while(queue.Count > 0){
            int size = queue.Count;
            for(int i = 0; i < size; i++){
                string word = queue.Dequeue();

                if(word == endWord) return changes;

                for(int j = 0; j < word.Length; j++){
                    // Se revisa cuáles cambios son palabras en la lista
                    for(char c = 'a'; c <= 'z'; c++){
                        char[] arr = word.ToCharArray();
                        arr[j] = c;

                        string str = new String(arr);
                        if(hs.Contains(str) && !visitedWords.Contains(str)){
                            queue.Enqueue(str);
                            visitedWords.Add(str);
                        }
                    }
                }
            }
            changes++;

        }
        return 0;
    }
}
