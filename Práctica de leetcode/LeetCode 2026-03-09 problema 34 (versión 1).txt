public class Solution {
    public int[] SearchRange(int[] nums, int target) {
        if(nums.Length == 0) return new int[] {-1, -1};
        
        return new int[]{BSFirst(nums, target), BSLast(nums, target)};
    }
    
    public int BSFirst(int[] nums, int target){
        int start = 0, end = nums.Length - 1;
        // Valor predeterminado
        int result = -1;

        while(start <= end){
            int mid = (start + end) / 2;
        
            if(nums[mid] == target){
                result = mid;
                // El objetivo todavía puede aparecer a la izquierda
                end = mid - 1;
            }
            else if(nums[mid] < target) start = mid + 1;
            else end = mid - 1;
        }
        
        return result;
    }
    
    public int BSLast(int[] nums, int target){
        int start = 0, end = nums.Length - 1;
        // Valor predeterminado
        int result = -1;

        while(start <= end){
            int mid = (start + end) / 2;
        
            if(nums[mid] == target){
                result = mid;
                // El objetivo todavía puede aparecer a la derecha
                start = mid + 1;
            }
            else if(nums[mid] < target) start = mid + 1;
            else end = mid - 1;
        }
        
        return result;
    }
}
