public class Solution {
    public int Jump(int[] nums) {
        int jumpEnd = 0, jumpRange = 0;
        int numJumps = 0;
        
        for(int i = 0; i < nums.Length - 1; i++){
            // Se obtiene la longitud del siguiente salto a realizar
            jumpRange = Math.Max(jumpRange, i + nums[i]);
            // Se completó el salto actual, así que aumenta la cantidad de saltos
            // y se empieza el siguiente salto
            if(i == jumpEnd){
                numJumps++;
                jumpEnd = jumpRange;    
            }
        }
        
        return numJumps;
    }
}
