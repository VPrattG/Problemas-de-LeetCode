public class Solution {
    public string CountAndSay(int n) {
        if(n == 1) return "1";
        
        string current = "1";

        for(int i = 1; i < n; i++){
            current = FindNextValue(current);
        }

        return current;
    }
    private string FindNextValue(string s){
        StringBuilder sb = new StringBuilder();
        int amount = 1;

        for(int i = 0; i < s.Length; i++){
            // Para distinguir cuando el siguiente valor es distinto o si es el fin de la cadena
            if(i == s.Length - 1 || s[i] != s[i + 1]){
                // Se agregan a la nueva cadena
                sb.Append(amount);
                sb.Append(s[i]);

                // Para empezar a contar el siguiente valor numérico
                amount = 1;
            }
            else{
                amount++;
            }
        }

        return sb.ToString();
    }
}
