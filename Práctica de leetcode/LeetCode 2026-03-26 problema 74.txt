public class Solution {
    public bool SearchMatrix(int[][] matrix, int target) {
        // Modificación ligera de búsqueda binaria
        int rows = matrix.Length, cols = matrix[0].Length;
        // rows * cols es como si se obtuviera el .Length de la matriz convertida en arreglo
        int left = 0, right = rows * cols - 1;

        while(left <= right){
            // Se necesita hacer esto porque no es un arreglo
            int mid = left + (right - left) / 2;
            int row = mid / cols;
            int col = mid % cols;
            int val = matrix[row][col];
            
            // Esta parte sigue siendo idéntica
            if(val == target) return true;
            else if(val < target) left = mid + 1;
            else right = mid - 1;
        }

        return false;
    }
}
