public class Solution {
    public int CanCompleteCircuit(int[] gas, int[] cost) {
        int totalGas = 0;
        int curGas = 0;
        int startPos = 0;
        
        for(int i = 0; i < gas.Length; i++){
            totalGas += gas[i] - cost[i];
            curGas += gas[i] - cost[i];

            if(curGas < 0){
                curGas = 0;
                startPos = i + 1;
            }
        }

        return totalGas >= 0 ? startPos : -1;
    }
}
