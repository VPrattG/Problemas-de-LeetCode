public class Solution {
    public bool ValidMountainArray(int[] arr) {
        if(arr.Length < 3) return false;
        
        int left = 0, right = arr.Length - 1;
        
        // left + 1 < arr.Length y right > 0 para no buscar en índices inexistentes
        while(left + 1 < arr.Length && arr[left] < arr[left + 1]) left++;
        while(right > 0 && arr[right - 1] > arr[right]) right--;
        
        // Las últimas dos condiciones son para no considerar arreglos no descendientes o
        // no ascendientes
        return left == right && left != 0 && right != arr.Length - 1;
    }
}
