public class Solution {
    private const int modulo = 1000000007;
    
    public int AssignEdgeWeights(int[][] edges) {
        int n = edges.Length + 2;

        List<List<int>> nodes = new List<List<int>>(n);
        for(int i = 0; i < n; i++) nodes.Add(new List<int>());

        foreach(int[] edge in edges){
            int u = edge[0];
            int v = edge[1];

            // Se agregan ambas posibilidades para probar los resultados
            nodes[u].Add(v);
            nodes[v].Add(u);
        }

        int maxDepth = AuxDFS(1, -1, nodes);

        // - 1 porque la mitad de las combinaciones son pares, y se buscan las impares
        // Entonces, (2^n) / 2 = 2^(n-1)
        return Power(2, maxDepth - 1);
    }

    private int AuxDFS(int node, int parent, List<List<int>> nodes){
        int result = 0;

        foreach(int child in nodes[node]){
            if(child != parent)
                result = Math.Max(result, 1 + AuxDFS(child, node, nodes));
        }

        return result;
    }

    private int Power(long num, long exp){
        long result = 1;

        while(exp > 0){
            if(exp % 2 > 0) result = result * num % modulo;
            num = num * num % modulo;
            exp /= 2;
        }

        return (int)result;
    }
}
