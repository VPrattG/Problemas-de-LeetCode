public class Solution {
    public int[] ProductExceptSelf(int[] nums) {
        int n = nums.Length;
        int prefix = 1, suffix = 1;
        int[] answer = new int[n];
        
        // Se pasa dos veces a través del arreglo
        // Sigue siendo 0(n) porque no son ciclos anidados
        for(int i = 0; i < n; i++){
            // El último valor tendrá el producto de los demás valores
            // Los de entremedio tendrán una porción del producto
            answer[i] = prefix;
            prefix *= nums[i];
        }
        for(int i = n - 1; i >= 0; i--){
            // El primer valor tendrá el producto de los demás valores
            // Los valores de entremedio recibirán la porción restante de sus productos
            answer[i] *= suffix;
            suffix *= nums[i];
        }

        return answer;
    }
}
