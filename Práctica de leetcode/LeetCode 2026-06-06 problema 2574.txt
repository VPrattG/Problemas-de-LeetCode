public class Solution {
    public int[] LeftRightDifference(int[] nums) {
        int prefix = 0;
        int suffix = 0;
        int n = nums.Length;
        int[] result = new int[n];
        
        for(int i = 0; i < n; i++){
            result[i] += prefix;
            prefix += nums[i];
        }

        for(int i = n - 1; i >= 0; i--){
            result[i] = Math.Abs(result[i] - suffix);
            suffix += nums[i];
        }

        return result;
    }
}
