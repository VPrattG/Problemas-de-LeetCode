public class Solution {
    public IList<string> RestoreIpAddresses(string s) {
        IList<string> result = new List<string>();

        if(s.Length > 12) return result;

        AuxBacktracking(s, 0, new List<string>(), result);
        return result;
    }

    private void AuxBacktracking(string s, int k, IList<string> list, IList<string> result){
        // k representa los valores que se han "visitado"
        // También revisa que haya 4 valores para juntar con .
        if(k == s.Length && list.Count == 4){
            string ipAddress = string.Join(".", list.ToArray());
            result.Add(ipAddress);
            return;
        }

        // i representa la cantidad de números que se van a revisar para la dirección IP
        for(int i = 1; i <= s.Length - k; i++){
            string sub = s.Substring(k, i);

            if(IsValid(sub)){
                list.Add(sub);
                AuxBacktracking(s, k + i, list, result);
                list.RemoveAt(list.Count - 1);
            }
        }
    }

    private bool IsValid(string sub){
        // En direcciones IP, cada valor individual no puede exceder 255
        // y 255 tiene una longitud de 3 caracteres
        if(sub.Length > 3) return false;
        
        int value = int.Parse(sub);
        if(value < 0 || value > 255 || sub.Length != value.ToString().Length) return false;

        return true;
    }
}
