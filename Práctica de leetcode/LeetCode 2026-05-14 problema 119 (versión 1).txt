public class Solution {
    public IList<int> GetRow(int rowIndex) {
        IList<int> row = new List<int>();
        row.Add(1); // Primer valor

        for(int i = 1; i <= rowIndex; i++){
            // Para manejar la siguiente fila
            IList<int> newRow = new List<int>();
            newRow.Add(1);
            
            // Omite el primer y último valor de la fila
            for(int j = 1; j < i; j++){
                int value = row[j - 1] + row[j];
                newRow.Add(value);
            }
                
            newRow.Add(1);
            
            row = newRow;
        }

        return row;
    }
}
