public class Solution {
    public int FindMin(int[] nums) {
        int start = 0, end = nums.Length - 1;

        while(start < end){
            // Por si no se rotó, o se rotó un múltiplo de n veces
            if(nums[start] < nums[end]) return nums[start];
            // Evita desbordamiento de datos
            int mid = start + (end - start) / 2;
            if(nums[start] <= nums[mid])
                start = mid + 1;
            else
                end = mid;
        }
        
        // El objetivo se encuentra al final del while
        return nums[start];
    }
}
