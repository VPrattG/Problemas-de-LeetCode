public class Solution {
    public int MinElement(int[] nums) {
        // 36 es el valor máximo por 9999
        // 9 + 9 + 9 + 9 = 36. 10000 solo se sumaría a 1
        int result = 36;
        
        foreach(int num in nums)
            result = Math.Min(result, num - 9 * ((num / 10) + (num/100) + (num/1000) + (num/10000)));
        
        return result;
    }
}
