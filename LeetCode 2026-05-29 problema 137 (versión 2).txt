public class Solution {
    public int SingleNumber(int[] nums) {
        int result = 0;

        for(int i = 0; i < 32; i++){
            int sum = 0;
            foreach(var num in nums){
                sum += (num >> i) & 1;
            }
            sum %= 3;
            result |= sum << i;
        }
        return result;
    }
}
